select prestamos.num_prestamo, prestamos.num_cliente, prestamos.fecha_inicio, DATE_ADD(prestamos.fecha_inicio, INTERVAL prestamos.num_cuotas DAY), prestamos.monto, prestamos.monto_final, prestamos.cuota, prestamos.num_cuotas, prestamos.orden_ruta, prestamos.cod_cartera, clientes.nombre, clientes.profesion, clientes.direccion, clientes.barrio, clientes.ciudad, clientes.celular, clientes.tel_fijo from prestamos, clientes where prestamos.num_cliente=clientes.num_cliente and prestamos.cod_cartera='3' and prestamos.estado='1'  order by orden_ruta
